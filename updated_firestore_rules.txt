rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users can read/write their own user document
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Forms and quizzes - authenticated users can create/update/delete their own
    match /forms/{formId} {
      // Allow public read access to published forms
      allow read: if resource.data.isPublished == true;
      // Allow authenticated users to read/write their own forms
      allow read, write: if request.auth != null && request.auth.uid == resource.data.createdBy;
      // Allow authenticated users to create new forms
      allow create: if request.auth != null && request.auth.uid == request.resource.data.createdBy;
    }
    
    // Responses - authenticated users can submit responses
    match /responses/{responseId} {
      // Allow authenticated users to create responses (including drafts)
      allow create: if request.auth != null && request.auth.uid == request.resource.data.userId;
      // Allow authenticated users to update their own responses (for drafts)
      allow update: if request.auth != null && request.auth.uid == resource.data.userId;
      // Allow authenticated users to delete their own responses (for cleaning up drafts)
      allow delete: if request.auth != null && request.auth.uid == resource.data.userId;
      // Allow form owners to read responses to their forms
      allow read: if request.auth != null && request.auth.uid == resource.data.formOwnerId;
      // Allow users to read their own responses
      allow read: if request.auth != null && request.auth.uid == resource.data.userId;
    }
    
    // Quiz responses - authenticated users can submit quiz responses
    match /quiz_responses/{responseId} {
      // Allow authenticated users to create quiz responses
      allow create: if request.auth != null && request.auth.uid == request.resource.data.userId;
      // Allow authenticated users to update their own responses (if needed)
      allow update: if request.auth != null && request.auth.uid == resource.data.userId;
      // Allow quiz owners to read responses to their quizzes
      allow read: if request.auth != null && request.auth.uid == resource.data.quizOwnerId;
      // Allow users to read their own responses
      allow read: if request.auth != null && request.auth.uid == resource.data.userId;
    }
    
    // Support tickets - authenticated users can create and manage their support tickets
    match /support_tickets/{ticketId} {
      // Allow authenticated users to create support tickets
      allow create: if request.auth != null && 
        request.auth.uid == request.resource.data.userId &&
        request.resource.data.keys().hasAll(['userId', 'userEmail', 'category', 'subject', 'description', 'status']);
      
      // Allow users to read their own support tickets
      allow read: if request.auth != null && request.auth.uid == resource.data.userId;
      
      // Allow users to update their own support tickets (for additional info)
      // But prevent changing critical fields like userId, createdAt
      allow update: if request.auth != null && 
        request.auth.uid == resource.data.userId &&
        request.resource.data.userId == resource.data.userId &&
        request.resource.data.createdAt == resource.data.createdAt;
      
      // Prevent deletion of support tickets for data integrity
      allow delete: if false;
      
      // Allow admins to read and update all support tickets (optional)
      // Uncomment and modify this if you have admin role checking
      // allow read, update: if request.auth != null && 
      //   request.auth.token.role == 'admin';
    }
  }
}
